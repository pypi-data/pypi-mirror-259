```
docker run -tid -p <port>:80 apachecn0/ruby-yuanma-jiedu
# 访问 http://localhost:{port} 查看文档
```

```
pip install ruby-yuanma-jiedu
ruby-yuanma-jiedu <port>
# 访问 http://localhost:{port} 查看文档
```

```
ruby-yuanma-jiedu <port>

```
