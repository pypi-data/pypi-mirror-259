```
docker pull apachecn0/spring-2-3-2-kaifa-jianming-jiaocheng-zhangyong
docker run -tid -p <port>:80 apachecn0/spring-2-3-2-kaifa-jianming-jiaocheng-zhangyong
# 访问 http://localhost:{port} 查看文档
```
