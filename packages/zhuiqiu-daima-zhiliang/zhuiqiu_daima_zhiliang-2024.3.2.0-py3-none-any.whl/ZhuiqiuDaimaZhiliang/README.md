```

docker run -tid -p <port>:80 apachecn0/zhuiqiu-daima-zhiliang
# 访问 http://localhost:{port} 查看文档
```

pip install zhuiqiu-daima-zhiliang
zhuiqiu-daima-zhiliang <port>
# 访问 http://localhost:{port} 查看文档
```

# 访问 http://localhost:{port} 查看文档
```
