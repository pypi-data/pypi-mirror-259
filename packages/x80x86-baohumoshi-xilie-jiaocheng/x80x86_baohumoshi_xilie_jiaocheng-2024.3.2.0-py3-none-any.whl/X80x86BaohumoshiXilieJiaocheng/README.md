```
docker pull apachecn0/x80x86-baohumoshi-xilie-jiaocheng
docker run -tid -p <port>:80 apachecn0/x80x86-baohumoshi-xilie-jiaocheng
# 访问 http://localhost:{port} 查看文档
```
